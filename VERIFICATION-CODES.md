# Google Site Verification
# Add your verification meta tag to index.html

# Instructions for Google Search Console:
# 1. Go to https://search.google.com/search-console
# 2. Add property for shashicaterers.com
# 3. Choose HTML tag verification method
# 4. Copy the meta tag and add it to index.html <head> section
# 5. Example: <meta name="google-site-verification" content="YOUR_VERIFICATION_CODE" />

# Bing Site Verification
# 1. Go to https://www.bing.com/webmasters
# 2. Add site shashicaterers.com
# 3. Choose HTML tag verification
# 4. Add to index.html: <meta name="msvalidate.01" content="YOUR_VERIFICATION_CODE" />

# Yandex Verification
# <meta name="yandex-verification" content="YOUR_VERIFICATION_CODE" />

# Pinterest Verification
# <meta name="p:domain_verify" content="YOUR_VERIFICATION_CODE" />

# Facebook Domain Verification
